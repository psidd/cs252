function daysShading(days, show_recent) {
	var ts = document.getElementsByClassName("engineTable");
	var i, j, dd_exists, dd_week;
	var dd_max = 0;
	for (i = 0; i < ts.length; i++) {
		var trs = ts[i].getElementsByTagName("tr");
		for (j = 0; j < trs.length; j++) {
			var dd = trs[j].getAttribute("data-days");
			if (dd) {
				dd_exists = 1;
				if (737655 - dd < days) {
					trs[j].style.backgroundColor = "#ccffcc";
					dd_week++;
				}
				else {
					trs[j].style.backgroundColor = null;
				}
				if (dd > dd_max)
					dd_max = dd;
			}
		}
	}
	if (!dd_week && dd_max && show_recent) {
		for (i = 0; i < ts.length; i++) {
			var trs = ts[i].getElementsByTagName("tr");
			for (j = 0; j < trs.length; j++) {
				var dd = trs[j].getAttribute("data-days");
				if (dd) {
					if (dd == dd_max)
						trs[j].style.backgroundColor = "#ccffcc";
					else
						trs[j].style.backgroundColor = null;
				}
			}
		}
	}
	
	if (!dd_exists)
		document.getElementById('shading-desc').style.display = 'none';
}
